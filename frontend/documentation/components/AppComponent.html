import {Component, OnDestroy, OnInit} from '@angular/core';
import {Router} from '@angular/router';

import {UserApi, LoopBackAuth} from 'shared/sdk/services';
import {UserMsgService} from 'shared/services/user-msg.service';

import { Subscription } from 'rxjs/Subscription';

import { LoopBackConfig } from 'shared/sdk';

import {environment} from '../environments/environment';

import {DatasetService} from 'datasets/dataset.service';
import {Store} from "@ngrx/store";
import * as ua from "state-management/actions/user.actions";
import * as dsa from "state-management/actions/datasets.actions";
@Component({
  selector : 'app-root',
  templateUrl : './app.component.html',
  styleUrls : [ './app.component.css' ],
  providers : [ UserApi ]
})
export class AppComponent implements OnDestroy, OnInit {
  title = 'CATANIE';
  us: UserApi;
  router: Router;
  username: string = null;
  msgSubscrip: Subscription;
  message = null;
  userSubscrip: Subscription;

  constructor(us: UserApi, router: Router, private uMsg: UserMsgService, private auth: LoopBackAuth, private store: Store<any>) {
    LoopBackConfig.setBaseURL(environment.lbBaseURL);
    console.log(LoopBackConfig.getPath());
    this.us = us;
    this.router = router;
    this.msgSubscrip = this.uMsg.getMessage().subscribe(msg => this.message = msg);
    this.store.select(state => state.root.user.currentUser).subscribe(current => {
        if(current && 'username' in current) {
            this.username = current['username'].replace('ms-ad.', '');
            if(!('realm' in current)) {
                this.store.dispatch({type: dsa.ADD_GROUPS, payload: this.username});
                this.store.dispatch({type: ua.ACCESS_USER_EMAIL, payload: this.username});
            }
        }else if('loggedOut' in current && current['loggedOut']) {
            if(window.location.pathname.indexOf('login') === -1) {
                window.location.replace('/login');
            }
        }
    });
  }


  /**
   * Handles initial check of username and updates
   * auth service (loopback does not by default)
   * @memberof AppComponent
   */
 ngOnInit() {
    this.store.dispatch({type: ua.RETRIEVE_USER});
  }

  ngOnDestroy() {
        // unsubscribe to ensure no memory leaks
        this.msgSubscrip.unsubscribe();
    }

  logout() {
    this.store.dispatch({type:ua.LOGOUT});
  }

  login() { this.router.navigateByUrl('/login'); }
}
