import {Location} from '@angular/common';
import {Component, Input, OnInit} from '@angular/core';
import {ActivatedRoute, NavigationExtras, Router} from '@angular/router';
import {Store} from "@ngrx/store";
import {DatasetService} from 'datasets/dataset.service';
import {OrigDatablock, RawDataset} from 'shared/sdk/models';
import {DatablockApi, RawDatasetApi} from 'shared/sdk/services';
import {UserMsgService} from 'shared/services/user-msg.service';
import * as dsa from "state-management/actions/datasets.actions";
import {DatasetState, initialState} from "state-management/state/dataset.store";

/**
 * Component to show details for a dataset, using the
 * form compoennt
 * @export
 * @class DatasetDetailComponent
 * @implements {OnInit}
 */
@Component({
  selector : 'dataset-detail',
  templateUrl : './dataset-detail.component.html',
  styleUrls : [ './dataset-detail.component.css' ]
})
export class DatasetDetailComponent implements OnInit {

  dataset: RawDataset = undefined;
  dataBlocks: Array<OrigDatablock>;
  error;

  constructor(private ds: RawDatasetApi, private db: DatablockApi,
              private cds: DatasetService, private route: ActivatedRoute,
              private uMsg: UserMsgService, private store: Store<any>) {

    this.store.select(state => state.root.datasets.currentSet)
        .subscribe(dataset => {
          if (dataset && Object.keys(dataset).length > 0) {
            this.dataset = <RawDataset>dataset;
            if (!('origdatablocks' in this.dataset)) {
              this.store.dispatch(
                  {type : dsa.DATABLOCKS, payload : this.dataset.pid});
            }
          }
        });
  };

  ngOnInit() {
    const self = this;
    this.route.params.subscribe(params => {
      this.store.dispatch({type : dsa.SEARCH_ID, payload : params.id});
    });
  }
}
