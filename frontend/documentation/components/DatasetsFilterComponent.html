import { Component, Input, OnInit } from '@angular/core';
import { DatasetService } from 'datasets/dataset.service';
import { TreeNode } from 'primeng/primeng';
import { AccessUserApi, RawDatasetApi } from 'shared/sdk/services';
import { ConfigService } from 'shared/services/config.service';

import { Store } from "@ngrx/store";
import { AppState, initialState } from "state-management/state/app.store";
import * as dStore from "state-management/state/dataset.store";
import * as dsa from "state-management/actions/datasets.actions";

@Component({
  selector: 'datasets-filter',
  templateUrl: './datasets-filter.component.html',
  styleUrls: ['./datasets-filter.component.css']
})
export class DatasetsFilterComponent implements OnInit {

  @Input() datasets: Array<any> = [];
  facets: Array<any> = [];
  months = [
    '', 'January', 'February', 'March', 'April', 'May', 'June', 'July', 'August',
    'September', 'October', 'November', 'December'
  ];
  yearChoices: TreeNode[];
  selectedTimes = [];
  startDate: Date;
  endDate: Date;

  dateCounts = {};
  dateFacet = [];

  location: {};
  locations = [];
  filteredLocations = [];

  group: {};
  groups = [];
  filteredGroups = [];
  date;

  filters;
  filterValues;

  constructor(private ds: RawDatasetApi, private cds: DatasetService, private store: Store<any>) {
    this.yearChoices = this.generateDates();
    const today = new Date();
    const start = new Date();
    if (!this.filterValues) {
     this.store.dispatch({ type: dsa.FILTER_UPDATE });
    }
  }

  updateDateLabels(dates, values) {
    for (let i = 0; i < dates.length; i++) {
      const date = dates[i];
      let index = -1;
      switch (date['type']) {
        case 'year': {
          index = values.findIndex(found => {
            if (date['data'] === found._id.year && !('month' in found._id) &&
              !('day' in found._id)) {
              return found;
            }
          });
          break;
        }
        case 'month': {
          index = values.findIndex(found => {
            if (date['data'] === found._id.month &&
              date['year'] === found._id.year && !('day' in found._id)) {
              return found;
            }
          });
          break;
        }
        case 'day': {
          index = values.findIndex(found => {
            if (date['data'] === found._id.day &&
              date['month'] === found._id.month &&
              date['year'] === found._id.year) {
              return found;
            }
          });
          break;
        }
      }
      if (index !== -1) {
        const split = date['label'].split(':');
        if (split.length > 0) {
          date['label'] = split[0];
        } else {
          date['label'] = date['label'];
        }
        date['label'] += ': ' + values[index].count;
      } else {
        date['label'] = date['label'].split(':')[0];
      }
      if ('children' in date) {
        this.updateDateLabels(date['children'], values);
      }
    }
  }

  /**
   * Load locations and ownergroups on start up and
   * only use unique values
   */
  ngOnInit() {
    this.store.select(state => state.root.datasets.activeFilters).subscribe(data => {
      this.filters = data;
      console.log(data);
    });
    this.store.select(state => state.root.datasets.filterValues).subscribe(values => {
      this.filterValues = values;
      if (this.filterValues) {
        if(this.filterValues['locations'] !== null) {
            this.locations = this.filterValues['locations'];
        }
        if(this.filterValues['groups'] !== null) {
            this.groups = this.filterValues['groups'];
        }
        this.dateFacet = [];
        let dates = [];
        if (this.filterValues['years']) {
          dates = dates.concat(this.filterValues['years']);
          dates.forEach(date => {
            const year = date['_id']['year'];
            const month = date['_id']['month'];
            let yid = { 'year': year };
            const yindex = dates.findIndex(found => {
              if (found._id.year === year && !('month' in found._id) &&
                !('day' in found._id)) {
                return found;
              }
            });
            if (yindex !== -1) {
              dates[yindex]['count'] += Number(date['count']);
            } else {
              dates.push({ '_id': yid, 'count': Number(date['count']) });
            }
            let mid = { 'year': year, 'month': month };
            const mindex = dates.findIndex(found => {
              if (found._id.year === year && found._id.month === month &&
                !('day' in found._id)) {
                return found;
              }
            });
            if (mindex !== -1) {
              dates[mindex]['count'] += Number(date['count']);
            } else {
              dates.push({ '_id': mid, 'count': Number(date['count']) });
            }
          });

          // this.yearChoices
          this.updateDateLabels(this.yearChoices, dates);
        }
      }
      //TODO subscribe to indfilterDatasetsvidual values and update date labels
    });

  }

  /**
   * Callback used whenever a time checkbox has been selected or
   * deselected
   * @param event
   */
  timeSelect(event) {

    if (this.selectedTimes.length > 0) {
      this.date = this.selectedTimes[0];
      const minYear = this.getMaxMinDate('year', false);
      const minMonth = this.getMaxMinDate('month', false, minYear);
      const minDay = this.getMaxMinDate('data', false, minYear, minMonth);
      this.filters.startDate = new Date(minYear + '/' + minMonth + '/' + minDay);
      const maxYear = this.getMaxMinDate('year', true);
      const maxMonth = this.getMaxMinDate('month', true, maxYear);
      const maxDay = this.getMaxMinDate('data', true, maxYear, maxMonth);
      this.filters.endDate = new Date(maxYear + '/' + maxMonth + '/' + maxDay);
      this.filters.endDate.setHours(23, 59, 59, 0);
    } else {
      this.filters.endDate = this.filters.startDate = undefined;
    }
    this.store.dispatch({ type: dsa.FILTER_UPDATE, payload: this.filters });
  }

  /**
   * Using the listeners for the selected time checkboxes,
   *
   * @param type
   * @param max
   * @param filterByYear
   * @param filterByMonth
   * @returns {null}
   */
  getMaxMinDate(type, max = true, filterByYear = -1, filterByMonth = -1) {
    let value = null;
    for (let i = 0; i < this.selectedTimes.length; i++) {
      if (this.selectedTimes[i]['type'] === 'day') {
        // declare value as first result for min/max
        if (value === null) {
          value = this.selectedTimes[i][type];
        }
        // check if value is less/more than
        if (max && value < this.selectedTimes[i][type] ||
          ((!max) && value > this.selectedTimes[i][type])) {
          // only save if filters are not applied or met
          if (filterByYear >= -1 &&
            this.selectedTimes[i]['year'] === filterByYear ||
            filterByYear === -1) {
            if (filterByMonth >= -1 &&
              this.selectedTimes[i]['month'] === filterByMonth ||
              filterByMonth === -1) {
              value = this.selectedTimes[i][type];
            }
          }
        }
      }
    }
    return value;
  }

  /**
   * Generate all the dates for the past <offset> years
   * into the format specified by the p-tree
   * @param {number} [offset=10]
   * @returns
   * @memberof DatasetsFilterComponent
   */
  generateDates(offset = 10) {
    const curYear = new Date().getFullYear();
    const years = [];
    for (let i = curYear; i >= curYear - offset; i--) {
      const year = {
        'label': '' + i,
        'data': i,
        'type': 'year',
        'children': this.genMonth(i)
      };
      years.push(year);
    }
    return years;
  }

  /**
   * Iterate and create an array of
   * all months in the year.
   * @param {any} year
   * @returns
   * @memberof DatasetsFilterComponent
   */
  genMonth(year) {
    const monthChildren = [];
    for (let m = 1; m < this.months.length; m++) {

      const month = {
        'data': m,
        'label': this.months[m],
        'type': 'month',
        'year': year,
        'children': this.genDaysInMonth(m, year)
      };
      monthChildren.push(month);
    }
    return monthChildren;
  }

  /**
   * Use Date object to calculate
   * the number of days in a month based on year.
   * @param {any} month
   * @param {any} year
   * @returns
   * @memberof DatasetsFilterComponent
   */
  genDaysInMonth(month, year) {
    const numDays = new Date(year, month, 0).getDate();
    const days = [];

    for (let d = 0; d < numDays; d++) {
      let label: string = '' + (d + 1);
      const day = {
        'data': d + 1,
        'label': label,
        'type': 'day',
        'month': month,
        'year': year
      };
      days.push(day);
    }
    return days;
  }

  /**
   * Handles search queries for the creation
   * location dropdown, this should link to a mongo filter query
   * @param {any} event
   * @memberof DatasetsFilterComponent
   */
  handleInputLocation(event) {
    this.filteredLocations =
      this.filterDatasets(event.query, 'creationLocation');
  }

  /**
   * Could be combined with above and handled based on event but
   * complicated to access source element through event
   * @param {any} event
   * @memberof DatasetsFilterComponent
   */
  handleInputOwner(event) {
    this.filteredGroups = this.filterDatasets(event.query, 'ownerGroup');
  }

  /**
   * Handles the event of the ngModel changing
   * Linked to both autocomplete values. Currently re runs the search when they
   * are cleared
   * @param {any} event
   * @memberof DatasetsFilterComponent
   */
  textValueChanged(event, key) {
    if (event.length === 0) {
      this.filters[key] = null;
    }else if (event.length >= 2) {
      if(key === 'groups') {
        this.filters[key] = [event];
      } else {
        this.filters[key] =  event;
      }
    }
    //TODO handle text values changing
    //TODO debounce time needs to be here even though it is in the effects?
    this.store.dispatch({ type: dsa.FILTER_UPDATE, payload: this.filters });
  }
  /**
   * Creates a filtered array based
   * on provided search term that matches the given key
   * @param {any} query
   * @param {any} key
   * @returns
   * @memberof DatasetsFilterComponent
   */
  filterDatasets(query, key) {
    const filtered = [];
    const array = key === 'creationLocation' ? this.locations : this.groups;
    if(array) {
      for (let i = 0; i < array.length; i++) {
        const loc = typeof (array[i]) === 'object' && !('_id' in array[i])
          ? array[i]
          : array[i]['_id'];
        if (loc && loc.toLowerCase().indexOf(query.toLowerCase()) === 0 &&
          filtered.indexOf(loc) === -1) {
          filtered.push(array[i]);
        }
      }
    }
    return filtered;
  }

  /**
   * Handle clicking of available locations
   */
  locSelected() {
    this.filters.creationLocation = this.location['_id'];
    this.store.dispatch({ type: dsa.FILTER_UPDATE, payload: this.filters });
  }

  /**
   * Handle clicking of available groups
   */
  groupSelected() {
    this.filters.groups = [this.group['_id']];
    console.log(this.filters);
    this.store.dispatch({ type: dsa.FILTER_UPDATE, payload: this.filters });
  }

  onSubmit() {
    this.store.dispatch({ type: dsa.FILTER_UPDATE, payload: this.filters });
  }

  // TODO add a clear action to the reducer?
  clearFacets() {
    this.selectedTimes = [];
    this.location = "";
    this.group = "";
    this.filters = dStore.initialState.filterValues;
    this.store.dispatch({ type: dsa.FILTER_UPDATE, payload: this.filters });
  }

   * Handle the dropdown click to show
   * a list of locations
   * @param {any} event
   * @memberof DatasetsFilterComponent
   */
  handleDropClick(event) {
    // TODO handle selected item
    console.log(event);
  }

  testSearch() { }
}
