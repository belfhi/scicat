import {
  Component,
  OnInit,
  Input
} from '@angular/core';
import {
  AccessUserApi,
  RawDatasetApi
} from 'shared/sdk/services';
import {
  ConfigService
} from 'shared/services/config.service';
import {TreeNode} from 'primeng/primeng';
import {DatasetService} from 'datasets/dataset.service';

@Component({
  selector: 'facet-box',
  templateUrl: './facet-box.component.html',
  styleUrls: ['./facet-box.component.css']
})
export class FacetBoxComponent implements OnInit {

  @Input() datasets: Array < any > = [];
  facets: Array<any> = [];
  months = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December'];
  yearChoices: TreeNode[];
  selectedTimes = [];

  location: string;
  locations = [];
  filteredLocations = [];

  group: string;
  groups = [];
  filteredGroups = [];

  constructor(private ds: RawDatasetApi, private cds: DatasetService) {
    // TODO auth error
    this.cds.datasetChange.subscribe(cmd => {
      this.datasets = this.cds.datasets;
    });
    // this.groupSrv.find().subscribe(groups => console.log(groups), error => console.log(error));
  }

  /**
   * Load locations and ownergroups on start up and
   * only use unique values
   */
  ngOnInit() {
    this.ds.find({limit: 1000}).subscribe(allSets => {
      allSets.map(set => {
        // TODO use AccessGroup API once access has been granted
        if (this.groups.indexOf(set['ownerGroup']) === -1) {
          this.groups.push(set['ownerGroup']);
        }
        if (this.locations.indexOf(set['creationLocation']) === -1) {
          this.locations.push(set['creationLocation']);
        }
      });
    }, error => {
      console.error(error);
    }, () => {
    });
    // if (this.datasets.length > 0 && this.datasets[0].constructor.name !== 'Object') {
    //   this.configSrv.getConfigFile(this.datasets[0].constructor.name).subscribe(res => {
    //       const config = res;
    //       for (const prop in config) {
    //         if (prop && config.hasOwnProperty(prop) && 'facet' in config[prop]) {
    //           this.facets.push(config[prop]);
    //         }
    //       }
    //     },
    //     error => {
    //       console.error(error);
    //     });
    // }
    this.yearChoices = this.generateDates();
  }

  timeSelect(event) {
    console.log(this.selectedTimes);
  }


  /**
   * Generate all the dates for the past <offset> years
   * into the format specified by the p-tree
   * @param {number} [offset=10]
   * @returns
   * @memberof FacetBoxComponent
   */
  generateDates(offset = 10) {
    const curYear = new Date().getFullYear();
    const years = [];
    for (let i = curYear; i >= curYear - offset; i--) {
      const year = {
        'label': i,
        'data': i,
        'children': this.genMonth(i)
      };
      years.push(year);
    }
    return years;
  }


  /**
   * Iterate and create an array of
   * all months in the year.
   * @param {any} year
   * @returns
   * @memberof FacetBoxComponent
   */
  genMonth(year) {
    const monthChildren = [];
    for (let m = 0; m < this.months.length; m++) {
      const month = {
        'data': this.months[m],
        'label': this.months[m],
        'children': this.genDaysInMonth(m + 1, year)
      };
      monthChildren.push(month);
    }
    return monthChildren;
  }


  /**
   * Use Date object to calculate
   * the number of days in a month based on year.
   * @param {any} month
   * @param {any} year
   * @returns
   * @memberof FacetBoxComponent
   */
  genDaysInMonth(month, year) {
    const numDays = new Date(year, month, 0).getDate();
    const days = [];
    for (let d  = 0; d < numDays; d++) {
      const day = {
        'data': d + 1,
        'label': d + 1
      };
      days.push(day);
    }
    return days;
  }


  /**
   * Handles search queries for the creation
   * location dropdown, this should link to a mongo filter query
   * @param {any} event
   * @memberof FacetBoxComponent
   */
  handleInputLocation(event) {
    this.filteredLocations = this.filterDatasets(event.query, 'creationLocation');
  }


  /**
   * Could be combined with above and handled based on event but
   * complicated to access source element through event
   * @param {any} event
   * @memberof FacetBoxComponent
   */
  handleInputOwner(event) {
    this.filteredGroups = this.filterDatasets(event.query, 'ownerGroup');
  }


  /**
   * Handles the event of the ngModel changing
   * Linked to both autocomplete values. Currently re runs the search when they are cleared
   * @param {any} event
   * @memberof FacetBoxComponent
   */
  autoCompleteValueChanged(event) {
    if (event.length === 0) {
      this.facetSearch();
    }
  }

  /**
   * Creates a filtered array based
   * on provided search term that matches the given key
   * @param {any} query
   * @param {any} key
   * @returns
   * @memberof FacetBoxComponent
   */
  filterDatasets(query, key) {
    // TODO filter server side
    const filtered = [];
    const array = key === 'creationLocation' ? this.locations : this.groups;
    for (let i = 0; i < array.length; i++ ) {
      const loc = array[i];
      if (loc.toLowerCase().indexOf(query.toLowerCase()) === 0 && filtered.indexOf(loc) === -1) {
        filtered.push(loc);
      }
    }
    return filtered;
  }

  /**
   * Handle clicking of available locations
   */
  locSelected() {
    this.facetSearch();
  }

  /**
   * Handle clicking of available groups
   */
  groupSelected() {
    this.facetSearch();
  }

  onSubmit() {
    this.facetSearch();
  }


  /**
   * Create loopback search query
   * for group and location params
   * Time has NOT been implemented yet
   * @memberof FacetBoxComponent
   */
  facetSearch() {
    const datasetSearch = {'limit': 100};
    if (this.group || this.location) {
      datasetSearch['where'] = {'and': []};
      const or: Array<object> = datasetSearch['where']['and'];
      if (this.group) {
        or.push({'ownerGroup': this.group});
      }
      if (this.location) {
        or.push({'creationLocation': this.location});
      }
    }
    this.cds.searchDatasets(datasetSearch);
  }


  /**
   * Handle the dropdown click to show
   * a list of locations
   * @param {any} event
   * @memberof FacetBoxComponent
   */
  handleDropClick(event) {
    // TODO handle selected item
    console.log(event);
  }

  testSearch() {


  }

}
